#include <stdio.h>

int main()
{
    long int n,i, rev = 0, digit;

    printf("Enter a number: ");
    scanf("%ld", &n);

    // Reverse number using for loop
    for(i=n; i> 0; i /= 10)
    {
        digit = i% 10;
        rev = rev * 10 + digit;
    }

    // Check palindrome
    if(n == rev)
        printf("It is a palindrome\n");
    else
        printf("It is not a palindrome\n");

    return 0;
}
