//only letters
#include <stdio.h>

int main()
{
    char str[9999];
    printf("Enter a string");
    scanf("%s",str);
    int i = 0;

    while (str[i] != '\0')
    {
        if ((str[i] >= 'a' && str[i] <= 'z') || 
            (str[i] >= 'A' && str[i] <= 'Z'))
        {
            printf("%c", str[i]);
        }
        i++;   
    }

    printf("\n");
    return 0;
}
