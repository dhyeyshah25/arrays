#include <stdio.h>

int main()
{
    int a,i,sum;
    sum=0;
    printf("Enter any number");
    scanf("%d",&a);
    for(i=1;i<=a/2;i++)
    {
	if(a%i==0)
{
        sum+=i;
}
    }
    if(sum==a)
    {
        printf("It is a perfect number");
    }
    else
        printf("it is not a perfect number");
}
