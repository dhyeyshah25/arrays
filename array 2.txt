//reverse an array
#include <stdio.h>

int main()
{
    int arr[5], arr2[5];
    int i, j;

    printf("enter five nnumbers");
    for(i = 0; i < 5; i++)
        scanf("%d", &arr[i]);

    // Reverse arr into arr2
    for(i = 0, j = 4; i < 5; i++, j--)
        arr2[i] = arr[j];

    // Print reversed array
    for(i = 0; i < 5; i++)
        printf("%d\n", arr2[i]);

    return 0;
}
